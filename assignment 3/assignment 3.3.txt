function check(st1,st2){
    var flag=0;
    if(st2.length==st1.length){
    for (var i=0;i<st1.length;i++){
        if(!st2.includes(st1[i]))
           {
            flag=1;
            break;}
    }}
    else
        flag=1;
    
    if(flag==1)
        console.log('no');
    else
        console.log('yes');

}

}
